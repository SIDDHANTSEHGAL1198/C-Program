

#include <stdio.h>

int main()
{
    int size;
    printf("Enter size of the stack\n");
    scanf("%d",&size);
    
    int stack[size];
    int top=-1;
    int data;
    int i;
    int choice;
    
    while(choice)
    {
        printf("Enter data\n");
        scanf("%d",&data);
        if(top==size-1)
        {
            printf("Overflow Condition\n");
            break;
        }
        else
        {
            top++;
            stack[top]=data;
        }
        printf("Do you want to continue(0 or 1)\n");
        scanf("%d",&choice);
    }
    printf("\nPrinting stack elements\n");
    for(i=top;i>=0;i--)
    {
        printf("%d ",stack[i]);
    }
    
    //Peek Operation
    int item=0;
    if(top==-1)
    {
        printf("Underflow Condition\n");
    }
    else
    {
        item=stack[top];
        top--;
    }
    
    printf("\nPrinting stack after peek operation\n");
    for(i=top;i>=0;i--)
    {
        printf("%d ",stack[i]);
    }
    
    return 0;
}
