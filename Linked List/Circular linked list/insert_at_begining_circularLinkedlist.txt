#include <malloc.h>

{
  struct node
  {

    struct node *next;
  };

  struct node *tail;

  while (choice)

      if (head == 0)

	  temp = head = newnode;

	{
	  temp->next = newnode;
	  temp = newnode;

      temp->next = head;
      printf ("Do u want to continue (0 or 1)\n");
      scanf ("%d", &choice);
    }

  temp = head;

  printf ("Printing Circular linked list\n");
  while (temp->next != head)
    {
      printf ("%d ", temp->data);
      temp = temp->next;
    }
  printf ("%d", temp->data);

  temp = head;
  while (temp->next != head)
    {
      tail = temp->next;

    }
